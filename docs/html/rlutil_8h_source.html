90  // Here be magic.
91  struct termios oldt, newt;
92  int ch;
93  tcgetattr(STDIN_FILENO, &oldt);
94  newt = oldt;
95  newt.c_lflag &= ~(ICANON | ECHO);
96  tcsetattr(STDIN_FILENO, TCSANOW, &newt);
97  ch = getchar();
98  tcsetattr(STDIN_FILENO, TCSANOW, &oldt);
99  return ch;
100 }

105 RLUTIL_INLINE int kbhit(void) {
106  // Here be dragons.
107  static struct termios oldt, newt;
108  int cnt = 0;
109  tcgetattr(STDIN_FILENO, &oldt);
110  newt = oldt;
111  newt.c_lflag &= ~(ICANON | ECHO);
112  newt.c_iflag = 0; // input mode
113  newt.c_oflag = 0; // output mode
114  newt.c_cc[VMIN] = 1; // minimum time to wait
115  newt.c_cc[VTIME] = 1; // minimum characters to wait for
116  tcsetattr(STDIN_FILENO, TCSANOW, &newt);
117  ioctl(0, FIONREAD, &cnt); // Read count
118  struct timeval tv;
119  tv.tv_sec = 0;
120  tv.tv_usec = 100;
121  select(STDIN_FILENO+1, NULL, NULL, NULL, &tv); // A small time delay
122  tcsetattr(STDIN_FILENO, TCSANOW, &oldt);
123  return cnt; // Return number of characters
124 }

363 RLUTIL_INLINE int getkey(void) {
364  #ifndef _WIN32
365  int cnt = kbhit(); // for ANSI escapes processing
366  #endif
367  int k = getch();
368  switch(k) {
369  case 0: {
370  int kk;
371  switch (kk = getch()) {
372  case 71: return KEY_NUMPAD7;
373  case 72: return KEY_NUMPAD8;
374  case 73: return KEY_NUMPAD9;
375  case 75: return KEY_NUMPAD4;
376  case 77: return KEY_NUMPAD6;
377  case 79: return KEY_NUMPAD1;
378  case 80: return KEY_NUMPAD2;
379  case 81: return KEY_NUMPAD3;
380  case 82: return KEY_NUMPAD0;
381  case 83: return KEY_NUMDEL;
382  default: return kk-59+KEY_F1; // Function keys
383  }}
384  case 224: {
385  int kk;
386  switch (kk = getch()) {
387  case 71: return KEY_HOME;
388  case 72: return KEY_UP;
389  case 73: return KEY_PGUP;
390  case 75: return KEY_LEFT;
391  case 77: return KEY_RIGHT;
392  case 79: return KEY_END;
393  case 80: return KEY_DOWN;
394  case 81: return KEY_PGDOWN;
395  case 82: return KEY_INSERT;
396  case 83: return KEY_DELETE;
397  default: return kk-123+KEY_F1; // Function keys
398  }}
399  case 13: return KEY_ENTER;
400 #ifdef _WIN32
401  case 27: return KEY_ESCAPE;
402 #else // _WIN32
403  case 155: // single-character CSI
404  case 27: {
405  // Process ANSI escape sequences
406  if (cnt >= 3 && getch() == '[') {
407  switch (k = getch()) {
408  case 'A': return KEY_UP;
409  case 'B': return KEY_DOWN;
410  case 'C': return KEY_RIGHT;
411  case 'D': return KEY_LEFT;
412  default: return -1;
413  }
414  } else return KEY_ESCAPE;
415  }
416 #endif // _WIN32
417  default: return k;
418  }
